async function generatePDFBlob(data){ const tpl = document.getElementById('pdf-template'); tpl.style.position = 'fixed'; tpl.style.left = '0'; tpl.style.top = '0'; tpl.style.width = '794px'; tpl.style.background = '#ffffff'; tpl.style.color = '#111111'; tpl.style.zIndex = '-1'; tpl.style.opacity = '1'; tpl.style.pointerEvents = 'none'; tpl.innerHTML = buildPDFHtml(data); await new Promise(resolve => setTimeout(resolve, 500)); const opt = { margin: [10,10,10,10], filename: `PasteRunner_${data.date || 'unknown'}_${data.shift || 'shift'}.pdf`, image: { type: 'jpeg', quality: 0.95 }, html2canvas: { scale: 2, useCORS: true, logging: false, backgroundColor: '#ffffff', scrollX: 0, scrollY: 0, windowWidth: 794 }, jsPDF: { unit: 'mm', format: 'a4', orientation: 'portrait' } }; const blob = await html2pdf() .set(opt) .from(tpl) .outputPdf('blob'); tpl.innerHTML = ''; tpl.style.left = '-9999px'; return blob; }
